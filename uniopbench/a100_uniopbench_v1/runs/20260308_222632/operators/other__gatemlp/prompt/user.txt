```py
"""Test script for GateMLP operator using the new optest framework.

This is the new unified test entry point that replaces check_cuda.py and
check_triton.py.
"""

import os
import sys
import argparse
from dataclasses import dataclass

import torch
from optest import (
    TestCase,
    TensorSpec,
    ScalarSpec,
    TestConfig,
    CUDABackend,
    TritonBackend,
    run_test,
    print_result,
)

from torch_.ref import torch_kernel

# Add parent directory to path for imports
TESTCASE_DIR = os.path.dirname(__file__)
sys.path.insert(0, TESTCASE_DIR)


@dataclass
class GateMLPParams:
    """GateMLP test parameters."""

    batch: int = 32
    k: int = 128
    n: int = 128


# Define test case using new framework
testcase = TestCase(
    tensor_specs=[
        TensorSpec("X", torch.float16, ("batch", "k"), "input"),
        TensorSpec("A", torch.float16, ("k", "n"), "input"),
        TensorSpec("B", torch.float16, ("k", "n"), "input"),
        TensorSpec("output", torch.float32, ("batch", "n"), "output"),
    ],
    scalar_specs=[
        ScalarSpec("batch", int, lambda p: p.batch),
        ScalarSpec("K", int, lambda p: p.k),
        ScalarSpec("N", int, lambda p: p.n),
    ],
    torch_kernel=torch_kernel,
)


def parse_args():
    """Parse command line arguments."""
    parser = argparse.ArgumentParser(description="GateMLP kernel test (new framework)")
    parser.add_argument(
        "--compile-only",
        action="store_true",
        help="Only compile the kernel without running tests",
    )
    parser.add_argument(
        "--no-perf", action="store_true", help="Disable performance testing"
    )
    parser.add_argument(
        "--backend",
        choices=["cuda", "triton"],
        default="cuda",
        help="Backend to test (default: cuda)",
    )
    return parser.parse_args()


def main():
    """Main test entry point."""
    args = parse_args()

    # Create test configuration
    config = TestConfig(
        enable_perf=not args.no_perf,
        compile_only=args.compile_only,
    )

    # Create backend
    if args.backend == "cuda":
        backend = CUDABackend()
    else:  # triton
        backend = TritonBackend()

    # Create test parameters
    params = GateMLPParams()

    # Run test
    print(f"\n{'=' * 80}")
    print(f"Testing GateMLP operator with {backend.name} backend")
    print(f"X: ({params.batch}, {params.k}), A/B: ({params.k}, {params.n})")
    print(f"Output: ({params.batch}, {params.n})")
    print(f"{'=' * 80}\n")

    result = run_test(
        testcase_dir=TESTCASE_DIR,
        testcase=testcase,
        backend=backend,
        params=params,
        config=config,
    )

    # Print results
    if not args.compile_only:
        print_result(result)

    return 0 if result.passed or result.skipped else 1


if __name__ == "__main__":
    sys.exit(main())

```

## File: torch_/ref.py

```py
import torch
import torch.nn.functional as F


def torch_kernel(x: torch.Tensor, a: torch.Tensor, b: torch.Tensor) -> torch.Tensor:
    x64 = x.float().double()
    o1 = torch.matmul(x64, a.float().double())
    o2 = torch.matmul(x64, b.float().double())
    return (F.silu(o1) * o2).float()
```

## File: get_data.py
```py
import torch
import ctypes
from dataclasses import dataclass
from optest.tools.checker import SEED


@dataclass
class Params:
    batch: int = 32
    k: int = 128
    n: int = 128


def get_cuda_argtypes():
    return [
        ctypes.c_void_p,  # X
        ctypes.c_void_p,  # A
        ctypes.c_void_p,  # B
        ctypes.c_void_p,  # output
        ctypes.c_int,  # batch
        ctypes.c_int,  # K
        ctypes.c_int,  # N
    ]


def get_cuda_torch_inputs(params: Params):
    torch.manual_seed(SEED)
    x = torch.randn((params.batch, params.k), dtype=torch.float16, device="cuda")
    a = torch.randn((params.k, params.n), dtype=torch.float16, device="cuda")
    b = torch.randn((params.k, params.n), dtype=torch.float16, device="cuda")
    output = torch.empty((params.batch, params.n), dtype=torch.float32, device="cuda")
    cuda_all_inputs = [x, a, b, output, params.batch, params.k, params.n]
    torch_all_inputs = [x, a, b]
    return cuda_all_inputs, torch_all_inputs, [output]


def cuda_output_tensor_transform(tensor):
    return tensor
```

## File: check_triton.py
```py
import os
import sys
import argparse
from get_data import get_cuda_torch_inputs, Params, cuda_output_tensor_transform
from torch_.ref import torch_kernel
from triton_.kernel import triton_kernel
from optest.tools.checker import compare_results

TESTCASE_ROOT_DIR = os.path.dirname(__file__)
sys.path.insert(0, TESTCASE_ROOT_DIR)


def parse_args():
    parser = argparse.ArgumentParser(
        description="gatemlp_32_128_128 Triton kernel test"
    )
    parser.add_argument(
        "--no-perf", action="store_true", help="Placeholder flag for compatibility"
    )
    return parser.parse_args()


def main():
    params = Params()
    _, torch_inputs, _ = get_cuda_torch_inputs(params)
    torch_output = cuda_output_tensor_transform(torch_kernel(*torch_inputs))
    triton_output = cuda_output_tensor_transform(triton_kernel(*torch_inputs))
    compare_results(torch_output, triton_output, test_type=["PyTorch", "Triton"])


if __name__ == "__main__":
    parse_args()
    main()
```
